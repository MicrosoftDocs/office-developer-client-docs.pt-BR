## <a name="definition"></a>Definição

```XML
          <xs:complexType name="Ellipse_Type">
          
          <xs:complexContent>
          <xs:extension base="GeometryRow_Type">
          <xs:sequence>
    <xs:element name="Cell"  type="Cell_Type"
     minOccurs="0"
     maxOccurs="unbounded"

    </xs:element>

      </xs:extension>
      </xs:complexContent>
      </xs:complexType>
      
```
